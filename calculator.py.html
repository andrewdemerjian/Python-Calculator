193      def button_analysis(self, num):
194  
195          # Any time AC is pressed the calculator should completely reset
196          if num == "AC":
197              self.reset(num)
198  
199          else:
200              # If statement checking two things before adding to the second number:
201              # If one of the operators has been entered previously
202              # If the entry is NOT an operator itself.
203              if (self.mult or self.add or self.div or self.sub) and (not num == " + " and
204                  not num == " - " and not num == " * " and not num == " / " and not num == " = "):
205                  self.numlist2.append(num)
206  
207              # Calls method to check if an operator has been entered
208              self.determine_operator(num)
209  
210              # Adds to the list of entered values as long as the equals sign isn't pressed.
211              # It will add an operator to the string as well if one hasn't been added already.
212              if self.shouldaddoperator and not num == " = ":
213                  self.numlist.append(num)
214  
215              # If the operator has already been added, it makes sure that none of the below operators are pressed
216              # before adding the entry to the string.
217              elif not num == " + " and not num == " - " and not num == " * " and not num == " / " and not num == " = ":
218                  self.numlist.append(num)
219  
220              # Creates and updates strings of the list of entered values
221              self.numstring = ''.join(self.numlist)
222              self.numstring2 = ''.join(self.numlist2)
223  
224              # Change the label to display the answer if the equal sign is entered
225              if num == " = " and self.shouldcompute:
226                  self.label.setText(str(self.ans))
227  
228                  # Reset the text strings and operator flags
229                  self.reset(num)
230  
231              # Displays the number string label of all the entered values
232              else:
233                  self.label.setText(self.numstring)

235      def determine_operator(self, operator):
236          # Operator actions only initialized if one has not been previously called
237          self.should_add_operator()
238          self.should_compute()
239  
240          if self.shouldaddoperator:
241              # Sets specific operator flag to true
242              if operator == " * ":
243                  self.mult = True
244              elif operator == " + ":
245                  self.add = True
246              elif operator == " - ":
247                  self.sub = True
248              elif operator == " / ":
249                  self.div = True
250  
251              # Closes off the first number for computing
252              self.num1 = int(self.numstring)
253  
254          # See if conditions are right to compute
255          elif operator == " = " and self.shouldcompute:
256              self.compute()
257  
258      def compute(self):
259          # Finishes the second number for computing
260          self.num2 = int(self.numstring2)
261  
262          # Does main four computations based on which operator flag was set to true
263          if self.mult:
264              self.ans = self.num1 * self.num2
265  
266          elif self.div:
267              self.ans = self.num1 / self.num2
268              # We round the answer to the nearest 4 decimal places
269              self.ans = round(self.ans, 4)
270  
271          elif self.add:
272              self.ans = self.num1 + self.num2
273  
274          elif self.sub:
275              self.ans = self.num1 - self.num2
276  
277      def reset(self, num):
278  
279          # Operator flags all set to false
280          self.mult, self.div, self.add, self.sub = False, False, False, False
281  
282          # The number lists get cleared out
283          self.numlist, self.numlist2 = [], []
284  
285          # The number strings are also cleared. These lines may be redundant and unnecessary,
286          # but it is nice to have all the necessary variables reset in this method.
287          self.numstring = ''.join(self.numlist)
288          self.numstring2 = ''.join(self.numlist2)
289  
290          # This method gets called in two places: When AC is pressed and when " = " is pressed.
291          # Doesn't clear the label text when " = " is pressed, since we probably want to see the answer!
292          if not num == " = ":
293              self.label.setText(self.numstring)

295      def should_add_operator(self):
296  
297          # Always set to false before the conditional statement
298          self.shouldaddoperator = False
299  
300          # Conditions for adding an operator
301          # If the length of the first string is greater than zero
302          # If an operator has not already been added
303          if not self.mult and not self.add and not self.div and not self.sub and len(self.numlist) > 0:
304              self.shouldaddoperator = True
305  
306      def should_compute(self):
307  
308          # Always set to false before the conditional statement
309          self.shouldcompute = False
310  
311          # Conditions for adding an equals
312          # If the second number list has any inputs
313          if len(self.numlist2) > 0:
314              self.shouldcompute = True
